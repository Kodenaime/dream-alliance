import React, { useState } from 'react';

function SingleCheckboxSelection() {
  const [selectedValue, setSelectedValue] = useState(null);

  const handleCheckboxChange = (value) => {
    setSelectedValue(value);
  };

  return (
    <div>
      <Checkbox label="Option 1" value="option1" checked={selectedValue === "option1"} onChange={handleCheckboxChange} />
      <Checkbox label="Option 2" value="option2" checked={selectedValue === "option2"} onChange={handleCheckboxChange} />
      <Checkbox label="Option 3" value="option3" checked={selectedValue === "option3"} onChange={handleCheckboxChange} />
    </div>
  );
}

function Checkbox({ label, value, checked, onChange }) {
  return (
    <div>
      <input type="checkbox" id={value} value={value} checked={checked} onChange={(e) => onChange(e.target.value)} />
      <label htmlFor={value}>{label}</label>
    </div>
  );
}

export default SingleCheckboxSelection;
